## 3. Database Schema (write this first — everything derives from it)

```ts
// src/server/db/schema.ts
import { pgTable, uuid, varchar, text, timestamp, pgEnum } from "drizzle-orm/pg-core";

export const referralStatusEnum = pgEnum("referral_status", ["new", "contacted", "scheduled", "cancelled"]);

export const referrals = pgTable("referrals", {
  id:                  uuid("id").defaultRandom().primaryKey(),
  // Patient
  patientFirstName:    varchar("patient_first_name", { length: 100 }).notNull(),
  patientLastName:     varchar("patient_last_name",  { length: 100 }).notNull(),
  dateOfBirth:         varchar("date_of_birth",      { length: 10  }).notNull(), // store as YYYY-MM-DD string
  phone:               varchar("phone",              { length: 20  }).notNull(),
  email:               varchar("email",              { length: 255 }),
  // Referring party
  lawFirmName:         varchar("law_firm_name",      { length: 255 }).notNull(),
  attorneyName:        varchar("attorney_name",      { length: 255 }).notNull(),
  attorneyEmail:       varchar("attorney_email",     { length: 255 }).notNull(),
  attorneyPhone:       varchar("attorney_phone",     { length: 20  }).notNull(),
  // Clinical
  primaryComplaint:    text("primary_complaint").notNull(),
  preferredLocation:   varchar("preferred_location", { length: 100 }).notNull(),
  appointmentType:     varchar("appointment_type",   { length: 20  }).notNull(), // "in-person" | "telemedicine"
  // Meta
  status:              referralStatusEnum("status").default("new").notNull(),
  createdAt:           timestamp("created_at").defaultNow().notNull(),
});

export type Referral     = typeof referrals.$inferSelect;
export type NewReferral  = typeof referrals.$inferInsert;
```

## 4. Shared Zod Schema (frontend + backend use the exact same file)

```ts
// src/lib/validations/referral.ts
import { z } from "zod";

export const CLINIC_LOCATIONS = [
  "Anaheim",
  "Culver City", 
  "Downey",
  "El Monte",
  "Long Beach",
  "Los Angeles",
] as const;

export const referralSchema = z.object({
  patientFirstName:  z.string().min(1, "First name is required").max(100),
  patientLastName:   z.string().min(1, "Last name is required").max(100),
  dateOfBirth:       z.string().regex(/^\d{4}-\d{2}-\d{2}$/, "Invalid date format"),
  phone:             z.string().min(7, "Phone number is required").max(20),
  email:             z.string().email("Invalid email").optional().or(z.literal("")),
  lawFirmName:       z.string().min(1, "Law firm name is required").max(255),
  attorneyName:      z.string().min(1, "Attorney name is required").max(255),
  attorneyEmail:     z.string().email("Invalid attorney email"),
  attorneyPhone:     z.string().min(7, "Attorney phone is required").max(20),
  primaryComplaint:  z.string().min(10, "Please describe the reason for referral").max(500),
  preferredLocation: z.enum(CLINIC_LOCATIONS, { errorMap: () => ({ message: "Select a location" }) }),
  appointmentType:   z.enum(["in-person", "telemedicine"]),
});

export type ReferralFormData = z.infer<typeof referralSchema>;
```

## 5. tRPC Router

```ts
// src/server/routers/referral.ts
import { z } from "zod";
import { publicProcedure, router } from "@/lib/trpc/server";
import { db } from "@/server/db";
import { referrals } from "@/server/db/schema";
import { referralSchema } from "@/lib/validations/referral";
import { desc, eq, ilike, or } from "drizzle-orm";

export const referralRouter = router({
  submit: publicProcedure
    .input(referralSchema)
    .mutation(async ({ input }) => {
      const [created] = await db.insert(referrals).values(input).returning({ id: referrals.id });

      return {
        id:             created.id,
        message:        "Referral submitted successfully.",
        followUpNote:   "Our team will contact the patient within 24 hours.",
      };
    }),

  // Bonus endpoint
  list: publicProcedure
    .input(z.object({
      search: z.string().optional(),
      status: z.enum(["new", "contacted", "scheduled", "cancelled"]).optional(),
    }).optional())
    .query(async ({ input }) => {
      const conditions = [];
      
      if (input?.search) {
        conditions.push(
          or(
            ilike(referrals.lawFirmName, `%${input.search}%`),
            ilike(referrals.patientFirstName, `%${input.search}%`),
            ilike(referrals.patientLastName, `%${input.search}%`),
          )
        );
      }
      if (input?.status) {
        conditions.push(eq(referrals.status, input.status));
      }

      return db
        .select()
        .from(referrals)
        .where(conditions.length ? and(...conditions) : undefined)
        .orderBy(desc(referrals.createdAt));
    }),
});
```

## 6. UI Design Direction

**Tone**: Clean medical-professional. Think Kaiser Permanente or a modern telehealth startup — not clinical-cold, not startup-flashy. Trust and clarity above everything.

**Color palette:**
- Primary: `#0f5298` (deep blue — medical trust)
- Accent: `#16a34a` (green — for success states)
- Neutral: slate-based grays
- Background: `#f8fafc`

**Form layout:**
- Two-column grid on desktop, single column on mobile
- Visually separated sections: "Patient Information" / "Referring Party" / "Appointment Details"
- `<fieldset>` / section headers with a subtle left border accent
- Sticky submit button area on mobile
- Character counter for the primaryComplaint textarea

**No elaborate animations.** One subtle fade-in on page load. Hover states on inputs. That's it.

## 8. Human-Touch Rules for Claude Code

These prevent the output from reading as AI-generated:

1. **No `// This function does X` comments** above every function. Comments only where the *why* isn't obvious.
2. **Pragmatic naming**: `lawFirmName` not `referringLawFirmOrganizationName`. Short but clear.
3. **Don't abstract prematurely**: No `createBaseFormField()` helpers for a single form. Just write the JSX.
4. **Realistic minor decisions**: Use `varchar` with explicit length limits in schema (not `text` for everything). This is what a careful engineer does.
5. **One or two intentional "I'd improve this in production" notes** in the README (e.g., date handling, rate limiting). This shows awareness without being defensive.
6. **No barrel exports (index.ts) for everything** — only add one if there are 3+ imports from same folder.
7. **Consistent but not religious spacing** — 2-space indent, no trailing semicolons debate (pick one and stick to it).
8. **The form should have a real-world loading state message** like "Submitting referral..." not "Loading...".
